二叉搜索树

什么是二叉搜索树？

二叉搜索树（Binary Search Tree），（又：二叉查找树，二叉排序树）作为一种经典的数据结构，它既有链表的快速插入与删除操作的特点，又有数组快速查找的优势，例如在文件系统和数据库系统一般会采用这种数据结构进行高效率的排序与检索操作。

二叉搜索树具有下列性质

  1. 若它的左子树不空，则左子树上所有结点的值均小于它的根结点的值。
  2. 若它的右子树不空，则右子树上所有结点的值均大于它的根结点的值。
  3. 它的左、右子树也分别为二叉搜索树。

二叉搜索树的主要操作有插入，查询，删除，遍历。

二叉搜索树的简单实现

public class BinarySearchTree {

    /**
     * 二叉搜索树的根
     */
    private Node root;
    /**
     * 节点数量
     */
    private int cnt;


    private BinarySearchTree() {
        this.cnt = 0;
    }

    /**
     * 获得根节点
     *
     * @return
     */
    public Node getRoot() {
        return root;
    }

    /**
     * 构建一个二叉搜索树
     *
     * @return
     */
    public static BinarySearchTree builder() {
        return new BinarySearchTree();
    }

    /**
     * 插入结点
     *
     * @param value 待插入节点的值
     * @return 根结点
     */
    public Node insert(int value) {
        // 构建一个节点
        Node curr = new Node(value);
        // 树为空，直接作为根结点
        if (root == null) {
            root = curr;
            return root;
        }
        // 保存空节点的父节点
        Node par = null;
        // 临时变量
        Node temp = root;
        // 寻求树中结点z的合适位置
        while (temp != null) {
            // 保存空节点的父节点
            par = temp;
            // 如果插入的值比当前节点小，则向它的左子树上面放，否则放到右子树
            temp = curr.key < temp.key ? temp.left : temp.right;
        }
        // 将插入的节点的父节点指向par
        curr.par = par;
        // 如果插入的值要比父节点的值小，将curr节点放置在par的左孩子上，否则放入右孩子上
        if (curr.key < par.key) {
            par.left = curr;
        } else {
            par.right = curr;
        }
        // 节点个数加一
        cnt++;
        return root;
    }


    /**
     * 中序遍历 即：排序结果
     *
     * @param root 二叉树的根节点
     * @param res  一个List集合 存放排序结果
     */
    public void inorder(Node root, List<Integer> res) {
        // 出口条件
        if (root == null) {
            return;
        }
        // 访问左子树
        inorder(root.left, res);
        // 在访问根节点
        res.add(root.key);
        // 访问右子树
        inorder(root.right, res);
    }

    /**
     * 寻找一个指定key的节点
     *
     * @param target
     * @return 返回第一个符合的节点
     */
    public Node search(int target) {
        // 存放结果节点，假设target在根节点
        Node ans = root;
        while (ans != null && ans.key != target) {
            // 如果插入的值比当前节点小，则向它的左子树上面放，否则放到右子树
            ans = target < ans.key ? ans.left : ans.right;
        }
        return ans;
    }

    /**
     * 获取指定根节点下的最小节点
     *
     * @param root 根节点
     * @return
     */
    public Node minimumNode(Node root) {
        // 存放结果节点，假设最小节点是根节点
        Node ans = root;
        // 根据二叉搜索树的性质，一直遍历当前节点的左孩子就可以了
        while (ans.left != null) {
            ans = ans.left;
        }
        return ans;
    }

    /**
     * 获取指定根节点下的最大节点
     *
     * @param root 根节点
     * @return
     */
    public Node largestNode(Node root) {
        // 存放结果节点，假设最大节点是根节点
        Node ans = root;
        // 根据二叉搜索树的性质，一直遍历当前节点的右孩子就可以了
        while (ans.right != null) {
            ans = ans.right;
        }
        return ans;
    }

    /**
     * 删除指定key的节点
     *
     * @param key
     * @return 返回根节点
     */
    public Node delete(int key) {
        // todo
        return root;
    }


    /**
     * 二叉搜索树的结点类
     */
    public static class Node {
        /**
         * 父结点
         */
        private Node par;
        /**
         * 左孩子
         */
        private Node left;
        /**
         * 右孩子
         */
        private Node right;
        /**
         * 节点值
         */
        private int key;

        public Node() {

        }

        public Node(int key) {
            this.key = key;
        }

        public void setPar(Node par) {
            this.par = par;
        }

        public Node getPar() {
            return par;
        }

        public void setKey(int key) {
            this.key = key;
        }

        public int getKey() {
            return key;
        }

        public void setLeft(Node left) {
            this.left = left;
        }

        public Node getLeft() {
            return left;
        }

        public void setRight(Node right) {
            this.right = right;
        }

        public Node getRight() {
            return right;
        }

    }


    public static void main(String[] args) {
        int[] temp = new int[]{1, 3, 2, 5, 4, 7, 6, 9, 8};
        BinarySearchTree bst = BinarySearchTree.builder();
        for (int i : temp) {
            bst.insert(i);
        }
        List<Integer> ans = new ArrayList<>();
        bst.inorder(bst.root, ans);
        ans.stream().forEach(e -> System.out.print(e + " "));
    }


}
