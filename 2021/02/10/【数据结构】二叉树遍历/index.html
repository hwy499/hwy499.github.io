【数据结构】二叉树遍历

本文最后更新于：3 天前

二叉树遍历

节点定义

public class TreeNode {
    char val;
    TreeNode left;
    TreeNode right;

    TreeNode() {
    }

    TreeNode(char val) {
        this.val = val;
    }

    TreeNode(char val, TreeNode left, TreeNode right) {
        this.val = val;
        this.left = left;
        this.right = right;
    }
}

前序遍历

动画演示

前序遍历首先访问根节点，然后遍历左子树，最后遍历右子树。

我们一起来看前序遍历的动画演示：

代码实现

递归式

public class BinaryTree {
    /**
     * 树根
     */
    TreeNode root;


    public BinaryTree(TreeNode root) {
        this.root = root;
    }

    /**
     * 前序遍历
     *
     * @return 返回前序遍历结果
     */
    public List<Character> preorderTraversal() {
        // 定义一个List集合存放前序遍历结果
        List<Character> res = new ArrayList<>();
        // 前序遍历
        preorder(root, res);
        return res;
    }

    /**
     * 前序遍历
     *
     * @param root 二叉树的根节点
     * @param res  一个List集合
     */
    public void preorder(TreeNode root, List<Character> res) {
        // 出口条件
        if (root == null) {
            return;
        }
        // 先访问根节点
        res.add(root.val);
        // 访问左子树
        preorder(root.left, res);
        // 访问右子树
        preorder(root.right, res);
    }


    public static void main(String[] args) {
        TreeNode H = new TreeNode('H');
        TreeNode I = new TreeNode('I',H,null);
        TreeNode G = new TreeNode('G', null, I);
        TreeNode C = new TreeNode('C');
        TreeNode E = new TreeNode('E');
        TreeNode D = new TreeNode('D', C, E);
        TreeNode A = new TreeNode('A');
        TreeNode B = new TreeNode('B', A, D);
        TreeNode F = new TreeNode('F', B, G);
        BinaryTree tree = new BinaryTree(F);
        System.out.print("前序遍历结果：");
        tree.preorderTraversal().forEach(item -> System.out.print(item + " "));
    }

}
// 运行结果：前序遍历结果：F B A D C E G I H

非递归式

public class BinaryTree {
    /**
     * 树根
     */
    TreeNode root;


    public BinaryTree(TreeNode root) {
        this.root = root;
    }

    /**
     * 先序遍历
     *
     * @return 返回先序遍历结果
     */
    public List<Character> preorderTraversal() {
        // 定义一个List集合存放先序遍历结果
        List<Character> res = new ArrayList<>();
        // 先序遍历
        preorder(root, res);
        return res;
    }

    /**
     * 先序遍历 非递归式
     *
     * @param root 二叉树的根节点
     * @param res  一个List集合
     */
    public void preorder(TreeNode root, List<Character> res) {
        // 用一个栈模拟函数的递归
        Stack<TreeNode> stack = new Stack<>();
        while (root != null || !stack.empty()) {
            // 如果当前节点不空
            while (root != null) {
                // 访问此节点
                res.add(root.val);
                // 将此节加入栈中，等待访问
                stack.push(root);
                // 访问左节点
                root = root.left;
            }
            // 如果栈不空 将栈顶弹出 并访问右节点
            if (!stack.empty()) {
                root = stack.pop();
                root = root.right;
            }
        }
    }


    public static void main(String[] args) {
        TreeNode H = new TreeNode('H');
        TreeNode I = new TreeNode('I', H, null);
        TreeNode G = new TreeNode('G', null, I);
        TreeNode C = new TreeNode('C');
        TreeNode E = new TreeNode('E');
        TreeNode D = new TreeNode('D', C, E);
        TreeNode A = new TreeNode('A');
        TreeNode B = new TreeNode('B', A, D);
        TreeNode F = new TreeNode('F', B, G);
        BinaryTree tree = new BinaryTree(F);
        System.out.print("先序遍历结果：");
        tree.preorderTraversal().forEach(item -> System.out.print(item + " "));
    }

}

     * 树根

层序遍历

动画演示

层序遍历就是逐层遍历树结构。

我们一起来看层序遍历的动画演示：

代码实现

广度优先搜索

public class BinaryTree {
    /**
     * 树根
     */
    TreeNode root;

    public BinaryTree(TreeNode root) {
        this.root = root;
    }

    /**
     * 层级遍历
     *
     * @return
     */
    public List<Character> levelOrder() {
        List<Character> ans = new ArrayList<>();
        if (root == null) {
            return ans;
        }
        // 维护一个队列，将未访问的元素向里面放置
        Queue<TreeNode> queue = new LinkedList<>();
        // 先把根节点放进去
        queue.offer(root);
        // 遍历队列
        while (!queue.isEmpty()) {
            // 获取此时的队列大小
            int currentLevelSize = queue.size();
            // 将此时队列中的数全部取出来
            for (int i = 1; i <= currentLevelSize; ++i) {
                // 取出队列前面的节点
                TreeNode node = queue.poll();
                // 将节点值加到结果集合里
                ans.add(node.val);
                // 左孩子不空 将其加入队列里
                if (node.left != null) {
                    queue.offer(node.left);
                }
                // 右孩子不空 将其加入队列里
                if (node.right != null) {
                    queue.offer(node.right);
                }
            }
        }
        return ans;
    }


    public static void main(String[] args) {
        TreeNode H = new TreeNode('H');
        TreeNode I = new TreeNode('I', H, null);
        TreeNode G = new TreeNode('G', null, I);
        TreeNode C = new TreeNode('C');
        TreeNode E = new TreeNode('E');
        TreeNode D = new TreeNode('D', C, E);
        TreeNode A = new TreeNode('A');
        TreeNode B = new TreeNode('B', A, D);
        TreeNode F = new TreeNode('F', B, G);
        BinaryTree tree = new BinaryTree(F);
        System.out.print("后序遍历结果：");
        tree.levelOrder().forEach(item -> System.out.print(item + " "));
    }
}
// 运行结果：后序遍历结果：F B G A D I C E H
