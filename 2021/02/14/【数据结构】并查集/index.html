【数据结构】并查集

本文最后更新于：8 分钟前

并查集学习

什么是并查集？

并查集是一种树型的数据结构，用于处理一些不相交集合（disjoint sets）的合并及查询问题。经典问题有：连通子图、最小生成树Kruskal算法等。

主要操作以及实现

初始化

把每个点所在集合初始化为其自身。

/**
 * 并查集数组
 */
private int par[];

/**
 * 初始化一个并查集
 *
 * @param n
 */
private void initSet(int n) {
    this.par = new int[n];
    for (int i = 0; i < n; i++) {
        par[i] = i;
    }
}

查找

查找元素所在的集合，即根节点。

/**
 * 寻找x所在的集合
 *
 * @param x
 * @return
 */
private int findSet(int x) {
    // 如果x 与 par[x] 不等说明 x 不是一个单独的集合，继续循环
    while (x != par[x]){
        x = par[x]; 
    }
    return x;
}

合并

将两个元素所在的集合合并为一个集合。

通常来说，合并之前，应先判断两个元素是否属于同一集合，这可用上面的“查找”操作实现。

/**
 * 将x集合与y集合合并
 *
 * @param x
 * @param y
 */
public void UnionSet(int x, int y) {
    // 寻找x所在的集合
    x = findSet(x);
    // 寻找y所在的集合
    y = findSet(y);
    // 如果x 和 y不是一个集合，就将x,y集合合并
    if (x != y) {
        // 将 x 所在集合合并到 y 所在集合
        par[x] = par[y];
    }
}

统计有多少个集合

在进行初始化时维护一个变量setNum用来存放集合个数，每当有集合进行合并的时候，就说明集合的总数少了一个。

/**
 * 集合数量
 */
private int setNum;

/**
 * 初始化一个并查集
 *
 * @param n
 */
private void initSet(int n) {
    this.par = new int[n];
    this.setNum = n;
    for (int i = 0; i < n; i++) {
        par[i] = i;
    }
}

/**
 * 将x集合与y集合合并
 *
 * @param x
 * @param y
 */
public void unionSet(int x, int y) {
    // 寻找x所在的集合
    x = findSet(x);
    // 寻找y所在的集合
    y = findSet(y);
    // 如果x 和 y不是一个集合，就将x,y集合合并
    if (x != y) {
        // 将 x 所在集合合并到 y 所在集合
        par[x] = par[y];
        // 每当发生合并时，就要将集合数量减一
        setNum--;
    }
}

/**
 * 获取集合数量
 *
 * @return
  */
public int getSetNum() {
    return setNum;
}

复杂度

查找函数findSet，合并函数unionSet的搜索深度是树的长度，复杂度都是O(N)，性能比较差。下面介绍合并和查询的优化方法，优化之后，查找和合并的复杂度都小于O(log2N)

并查集的优化

合并函数的优化

/**
 * 将x集合与y集合合并
 *
 * @param x
 * @param y
 */
public void unionSet(int x, int y) {
    // 寻找x所在的集合
    x = findSet(x);
    // 寻找y所在的集合
    y = findSet(y);
    // 如果x 和 y不是一个集合，就将x,y集合合并
    if (x != y) {
        // 将 x 所在集合合并到 y 所在集合
        par[x] = par[y];
        // 集合数量减一
        setNum--;
    }
}

在上述函数合并x,y时，需要搜到他们的根节点a,b，然后合并a,b,即把a的根节点改为b节点。a,b两个节点的高度不一样，如果把高度低的根节点改为高度高的根节点，这样会提升复杂度，但是如果把高度高的根节点改为高度低的根节点，降低了树的高度，就会降低复杂度。但是如何改呢？

我们可以在初始化的时候在额外初始化一个数组height用于保存每个节点当前所在树的高度。

/**
 * 高度数组
 */
 private int height[];
/**
 * 初始化一个并查集
 *
 * @param n
 */                
private void initSet(int n) {
    this.setNum = n;
    this.par = new int[n];
    this.height = new int[n];
    for (int i = 0; i < n; i++) {
        this.par[i] = i;
        // 初始化每个节点树的高度
        this.height[i] = 0;
    }
}

初始化完每个节点所在树的高度后，我们接下来在合并的函数中使用以及维护height数组。

/**
 * 将x集合与y集合合并
 *
 * @param x
 * @param y
 */                
public void unionSet(int x, int y) {
    // 寻找x所在的集合
    x = findSet(x);
    // 寻找y所在的集合
    y = findSet(y);
    // 如果x 和 y是一个集合，不做处理
    if (x == y) {
        return;
    }
    // 如果x所在树的高度小于y所在树的高度
    if (height[x] < height[y]) {
        // 把 x 的集合与 y的集合并拢，并把 y 当做祖先结点
        par[x] = y;
    } else {
        // 把 y 的集合与 x的集合并拢，并把 x 当做祖先结点
        par[y] = x;
        // 如果合并时两个节点所在的数一般高，则把其中一个高度加一
        if (height[x] == height[y]) {
            height[x]++;
        }
    }
    // 每当发生合并时，就要将集合数量减一
    setNum --;
}

到此合并函数的优化已经完毕。

查找函数的优化

/**
 * 寻找x所在的集合
 *
 * @param x
 * @return
 */
private int findSet(int x) {
    // 如果x 与 par[x] 不等说明 x 不是一个单独的集合，继续循环
    while (x != par[x]){
        x = par[x]; 
    }
    return x;
}

在上面的函数中寻找一个节点x的根节点时，如果par[x] != x ,这时我们需要做如下操作

  1. x = par[x]
  2. 接着判断x 与par[x]的关系
  3. 如果x != par[x]转到第一步，否则就返回x即可

如果按照上面的方法寻找根节点，循环的次数会随着树的高度增加而增加。如果我们在寻找到根节点后把树上所有的节点全指向根节点，如下图，那么下次搜索时就能在O(1)的时间内得到结果。

代码实现（递归式）

/**
 * 寻找x所在的集合
 *
 * @param x
 * @return
 */
private int findSet(int x) {
    // 如果x 与 par[x] 不等说明 x 不是一个单独的集合，继续循环
    if (x != par[x]) {
        // 这步很重要，如果找到了根节点后，由于采用的是递归，会在找到后，
        // 将这个节点值返回来从而达到修改子节点值的目的。
        par[x] = findSet(par[x]);
    }
    return par[x];
}

如果担心递归次数太多导致爆栈，则可以考虑用非递归的方式：

/**
 * 寻找x所在的集合
 *
 * @param x
 * @return
 */
private int findSet(int x) {
    int temp = x;
    // 如果x 与 par[x] 不等说明 x 不是一个单独的集合，
    while (temp != par[temp]) {
        temp = par[temp];
    }
    int i = x, j;
    // 将 x 节点所在的树子节点全部指向根节点temp
    while (i != temp) {
        // 先把当前节点的值，保留下来
        j = par[i];
        // 将根节点的值赋给当前节点
        par[i] = temp;
        // 将当前节点的值赋给i
        i = j;
    }
    return temp;
}

优化后的并查集

/**
 * 并查集数据结构实现
 */
public class DisjointSetUnion {

    /**
     * 并查集数组
     */
    private int par[];

    /**
     * 集合数量
     */
    private int setNum;

    /**
     * 高度数组
     */
    private int height[];

    /**
     * 构造函数
     *
     * @param n
     */
    public DisjointSetUnion(int n) {
        this.initSet(n);
    }


    /**
     * 初始化一个并查集
     *
     * @param n
     */
    private void initSet(int n) {
        this.setNum = n;
        this.par = new int[n];
        this.height = new int[n];
        for (int i = 0; i < n; i++) {
            this.par[i] = i;
            // 初始化每个节点树的高度
            this.height[i] = 0;
        }
    }

    /**
     * 寻找x所在的集合
     *
     * @param x
     * @return
     */
    private int findSet(int x) {
        // 如果x 与 par[x] 不等说明 x 不是一个单独的集合，继续循环
        if (x != par[x]) {
            // 这步很重要，如果找到了根节点后，由于采用的是递归，会在找到后，
            // 将这个节点值返回来从而达到修改子节点值的目的。
            par[x] = findSet(par[x]);
        }
        return par[x];
    }


    /**
     * 将x集合与y集合合并
     *
     * @param x
     * @param y
     */
    public void unionSet(int x, int y) {
        // 寻找x所在的集合
        x = findSet(x);
        // 寻找y所在的集合
        y = findSet(y);
        // 如果x 和 y是一个集合，不做处理
        if (x == y) {
            return;
        }
        // 如果x所在树的高度小于y所在树的高度
        if (height[x] < height[y]) {
            // 把 x 的集合与 y的集合并拢，并把 y 当做祖先结点
            par[x] = y;
        } else {
            // 把 y 的集合与 x的集合并拢，并把 x 当做祖先结点
            par[y] = x;
            // 如果合并时两个节点所在的数一般高，则把其中一个高度加一
            if (height[x] == height[y]) {
                height[x]++;
            }
        }
        // 每当发生合并时，就要将集合数量减一
        setNum--;
    }

    /**
     * 获取集合数量
     *
     * @return
     */
    public int getSetNum() {
        return setNum;
    }

}
